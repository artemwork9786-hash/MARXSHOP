# WebSocket: Динамический статус аккаунтов в реальном времени

## Решение: WebSocket

### Сервер (server/server.js)

1. Установить `ws` пакет
2. Создать WebSocket сервер на том же порту что и Express:
```js
const http = require("http");
const { WebSocketServer } = require("ws");

const server = http.createServer(app);
const wss = new WebSocketServer({ server });

// При изменении busyUntil:
function broadcastAccountUpdate(accountId, busyUntil) {
  const msg = JSON.stringify({ type: "account:update", accountId, busyUntil });
  wss.clients.forEach(client => {
    if (client.readyState === 1) client.send(msg);
  });
}

server.listen(PORT, "0.0.0.0");
```
3. Вызывать `broadcastAccountUpdate()` в endpoints где меняется busyUntil:
   - POST /api/create-order (подтверждение оплаты)
   - POST /api/cancel-order (отмена)

### Клиент (src/App.jsx)

1. Подключиться к WebSocket при монтировании:
```js
useEffect(() => {
  const ws = new WebSocket(API_URL.replace("http", "ws"));
  ws.onmessage = (e) => {
    const data = JSON.parse(e.data);
    if (data.type === "account:update") {
      setAccounts(prev => prev.map(a =>
        a.id === data.accountId ? { ...a, busyUntil: data.busyUntil } : a
      ));
    }
  };
  ws.onclose = () => setTimeout(() => { /* reconnect */ }, 3000);
  return () => ws.close();
}, []);
```
